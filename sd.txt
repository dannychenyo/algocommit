function twoNumberSum(array, targetSum) {
  array.sort((a, b) => a - b)
  let left = 0
  let right = array.length - 1
  while (left < right){
    let currentSum = array[left] + array[right]
    if(currentSum === targetSum){  
      return [left, right]
    } else if(currentSum < targetSum){
        left++
    } else if (currentSum > targetSum){
        right--
    } 
  }
}
